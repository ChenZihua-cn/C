快速排序（必会！） 

void quickSort(int arr[], int left, int right) {
    if (left >= right) return;
    
    int pivot = arr[left];  // 选第一个为基准
    int i = left, j = right;
    
    while (i < j) {
        while (i < j && arr[j] >= pivot) j--;
        while (i < j && arr[i] <= pivot) i++;
        if (i < j) swap(&arr[i], &arr[j]);
    }
    
    arr[left] = arr[i];
    arr[i] = pivot;
    
    quickSort(arr, left, i - 1);
    quickSort(arr, i + 1, right);
}

归并排序 

void merge(int arr[], int l, int m, int r) {
    int i, j, k;
    int n1 = m - l + 1, n2 = r - m;
    int L[n1], R[n2];
    
    for (i = 0; i < n1; i++) L[i] = arr[l + i];
    for (j = 0; j < n2; j++) R[j] = arr[m + 1 + j];
    
    i = 0; j = 0; k = l;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) arr[k++] = L[i++];
        else arr[k++] = R[j++];
    }
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

void mergeSort(int arr[], int l, int r) {
    if (l < r) {
        int m = l + (r - l) / 2;
        mergeSort(arr, l, m);
        mergeSort(arr, m + 1, r);
        merge(arr, l, m, r);
    }
}

堆排序 

// 调整堆
void heapify(int arr[], int n, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;
    
    if (left < n && arr[left] > arr[largest]) largest = left;
    if (right < n && arr[right] > arr[largest]) largest = right;
    
    if (largest != i) {
        swap(&arr[i], &arr[largest]);
        heapify(arr, n, largest);
    }
}

void heapSort(int arr[], int n) {
    // 建堆
    for (int i = n / 2 - 1; i >= 0; i--)
        heapify(arr, n, i);
    
    // 一个个取出元素
    for (int i = n - 1; i > 0; i--) {
        swap(&arr[0], &arr[i]);
        heapify(arr, i, 0);
    }
}

二分查找 

// 返回索引，找不到返回-1
int binarySearch(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;  // 防溢出
        if (arr[mid] == target) return mid;
        if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}

// 找左边界（第一个≥target的位置）
int lowerBound(int arr[], int n, int target) {
    int left = 0, right = n;
    while (left < right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] < target) left = mid + 1;
        else right = mid;
    }
    return left;
}

// 找右边界（第一个>target的位置）
int upperBound(int arr[], int n, int target) {
    int left = 0, right = n;
    while (left < right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] <= target) left = mid + 1;
        else right = mid;
    }
    return left;
}

ListNode* reverseList(ListNode *head) {
    ListNode *prev = NULL;
    ListNode *cur = head;
    
    while (cur) {
        ListNode *next = cur->next;  // 保存下一个
        cur->next = prev;             // 反转指向
        prev = cur;                   // 前移prev
        cur = next;                   // 前移cur
    }
    return prev;  // 新的头节点
}

链表反转（递归） 

ListNode* reverseListRecursive(ListNode *head) {
    if (!head || !head->next) return head;
    
    ListNode *newHead = reverseListRecursive(head->next);
    head->next->next = head;
    head->next = NULL;
    return newHead;
}

ListNode* findMiddle(ListNode *head) {
    ListNode *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
    }
    return slow;  // 中点
}

int hasCycle(ListNode *head) {
    ListNode *slow = head, *fast = head;
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        if (slow == fast) return 1;  // 有环
    }
    return 0;
}

哈希表

拉链法实现 

#include <stdlib.h>

typedef struct Node {
    int key;
    int value;
    struct Node *next;
} Node;

typedef struct {
    Node **buckets;
    int size;
} HashTable;

// 哈希函数
int hash(int key, int size) {
    return ((key % size) + size) % size;  // 处理负数
}

// 创建
HashTable* createHashTable(int size) {
    HashTable *ht = malloc(sizeof(HashTable));
    ht->size = size;
    ht->buckets = calloc(size, sizeof(Node*));
    return ht;
}

// 插入/更新
void put(HashTable *ht, int key, int value) {
    int idx = hash(key, ht->size);
    
    Node *cur = ht->buckets[idx];
    while (cur) {
        if (cur->key == key) {
            cur->value = value;  // 更新
            return;
        }
        cur = cur->next;
    }
    
    // 头插法
    Node *newNode = malloc(sizeof(Node));
    newNode->key = key;
    newNode->value = value;
    newNode->next = ht->buckets[idx];
    ht->buckets[idx] = newNode;
}

// 查找
int get(HashTable *ht, int key, int *found) {
    int idx = hash(key, ht->size);
    Node *cur = ht->buckets[idx];
    
    while (cur) {
        if (cur->key == key) {
            *found = 1;
            return cur->value;
        }
        cur = cur->next;
    }
    *found = 0;
    return -1;
}

// 删除
void delete(HashTable *ht, int key) {
    int idx = hash(key, ht->size);
    Node *cur = ht->buckets[idx];
    Node *prev = NULL;
    
    while (cur) {
        if (cur->key == key) {
            if (prev) prev->next = cur->next;
            else ht->buckets[idx] = cur->next;
            free(cur);
            return;
        }
        prev = cur;
        cur = cur->next;
    }
}

Dijkstra最短路 

#include <limits.h>

#define INF INT_MAX

void dijkstra(int graph[MAX_N][MAX_N], int n, int start) {
    int dist[MAX_N];
    bool visited[MAX_N] = {false};
    
    for (int i = 0; i < n; i++) dist[i] = INF;
    dist[start] = 0;
    
    for (int i = 0; i < n; i++) {
        // 找未访问的最小距离点
        int u = -1, minDist = INF;
        for (int j = 0; j < n; j++) {
            if (!visited[j] && dist[j] < minDist) {
                minDist = dist[j];
                u = j;
            }
        }
        
        if (u == -1) break;
        visited[u] = true;
        
        // 更新邻居
        for (int v = 0; v < n; v++) {
            if (graph[u][v] && !visited[v] && dist[u] + graph[u][v] < dist[v]) {
                dist[v] = dist[u] + graph[u][v];
            }
        }
    }
}

#include <string.h>

// 计算next数组
void computeLPS(char *pattern, int m, int *lps) {
    int len = 0;
    lps[0] = 0;
    int i = 1;
    
    while (i < m) {
        if (pattern[i] == pattern[len]) {
            len++;
            lps[i] = len;
            i++;
        } else {
            if (len != 0) len = lps[len - 1];
            else {
                lps[i] = 0;
                i++;
            }
        }
    }
}

// KMP搜索
void kmpSearch(char *text, char *pattern) {
    int n = strlen(text);
    int m = strlen(pattern);
    int lps[m];
    
    computeLPS(pattern, m, lps);
    
    int i = 0, j = 0;
    while (i < n) {
        if (pattern[j] == text[i]) {
            i++;
            j++;
        }
        
        if (j == m) {
            printf("Pattern found at index %d\n", i - j);
            j = lps[j - 1];
        } else if (i < n && pattern[j] != text[i]) {
            if (j != 0) j = lps[j - 1];
            else i++;
        }
    }
}

动态规划（DP）

核心思想

DP = 大问题拆成小问题 + 记忆化存储 + 状态转移

模板1：斐波那契/爬楼梯 

// 基础版
int fib(int n) {
    if (n <= 1) return n;
    int dp[n + 1];
    dp[0] = 0; dp[1] = 1;
    for (int i = 2; i <= n; i++)
        dp[i] = dp[i-1] + dp[i-2];
    return dp[n];
}

// 空间优化版（滚动数组）
int fibOptimized(int n) {
    if (n <= 1) return n;
    int prev2 = 0, prev1 = 1;
    for (int i = 2; i <= n; i++) {
        int curr = prev1 + prev2;
        prev2 = prev1;
        prev1 = curr;
    }
    return prev1;
}

模板2：0/1背包 

// n个物品，背包容量W，每个物品有价值v和重量w
// 求最大价值
int knapsack(int W, int wt[], int val[], int n) {
    int dp[W + 1];
    memset(dp, 0, sizeof(dp));
    
    for (int i = 0; i < n; i++) {
        // 倒序遍历容量（0/1背包）
        for (int w = W; w >= wt[i]; w--) {
            dp[w] = max(dp[w], dp[w - wt[i]] + val[i]);
        }
    }
    return dp[W];
}

模板3：最长递增子序列（LIS） 

int lengthOfLIS(int nums[], int n) {
    if (n == 0) return 0;
    
    int dp[n];
    int maxLen = 1;
    
    for (int i = 0; i < n; i++) {
        dp[i] = 1;  // 每个元素至少长度为1
        for (int j = 0; j < i; j++) {
            if (nums[j] < nums[i]) {
                dp[i] = max(dp[i], dp[j] + 1);
            }
        }
        maxLen = max(maxLen, dp[i]);
    }
    return maxLen;
}

// 二分优化版 O(nlogn)
int lengthOfLISOptimized(int nums[], int n) {
    int tails[n];
    int size = 0;
    
    for (int i = 0; i < n; i++) {
        int left = 0, right = size;
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (tails[mid] < nums[i]) left = mid + 1;
            else right = mid;
        }
        tails[left] = nums[i];
        if (left == size) size++;
    }
    return size;
}

模板4：最长公共子序列（LCS） 

int longestCommonSubsequence(char *text1, char *text2) {
    int m = strlen(text1);
    int n = strlen(text2);
    
    // dp[i][j] = text1[0..i-1]和text2[0..j-1]的LCS长度
    int dp[m + 1][n + 1];
    memset(dp, 0, sizeof(dp));
    
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (text1[i-1] == text2[j-1])
                dp[i][j] = dp[i-1][j-1] + 1;
            else
                dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
        }
    }
    return dp[m][n];
}

模板5：编辑距离 

int minDistance(char *word1, char *word2) {
    int m = strlen(word1);
    int n = strlen(word2);
    
    int dp[m + 1][n + 1];
    
    // 初始化边界
    for (int i = 0; i <= m; i++) dp[i][0] = i;
    for (int j = 0; j <= n; j++) dp[0][j] = j;
    
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (word1[i-1] == word2[j-1])
                dp[i][j] = dp[i-1][j-1];
            else
                dp[i][j] = 1 + min(dp[i-1][j-1],    // 替换
                              min(dp[i-1][j],       // 删除
                                  dp[i][j-1]));     // 插入
        }
    }
    return dp[m][n];
}

模板：区间求和 + 单点修改 

#define MAXN 100000

int tree[4 * MAXN];  // 4倍空间
int arr[MAXN];

// 建树 [l, r]
void build(int node, int l, int r) {
    if (l == r) {
        tree[node] = arr[l];
        return;
    }
    int mid = (l + r) / 2;
    build(node * 2, l, mid);
    build(node * 2 + 1, mid + 1, r);
    tree[node] = tree[node * 2] + tree[node * 2 + 1];
}

// 单点修改：把idx位置的值改为val
void update(int node, int l, int r, int idx, int val) {
    if (l == r) {
        tree[node] = val;
        return;
    }
    int mid = (l + r) / 2;
    if (idx <= mid)
        update(node * 2, l, mid, idx, val);
    else
        update(node * 2 + 1, mid + 1, r, idx, val);
    tree[node] = tree[node * 2] + tree[node * 2 + 1];
}

// 区间查询 [ql, qr]
int query(int node, int l, int r, int ql, int qr) {
    if (ql <= l && r <= qr)  // 完全包含
        return tree[node];
    if (r < ql || l > qr)    // 完全不包含
        return 0;
    
    int mid = (l + r) / 2;
    int leftSum = query(node * 2, l, mid, ql, qr);
    int rightSum = query(node * 2 + 1, mid + 1, r, ql, qr);
    return leftSum + rightSum;
}

// 使用示例
void segmentTreeExample() {
    int n = 6;
    arr[1] = 1; arr[2] = 3; arr[3] = 5;
    arr[4] = 7; arr[5] = 9; arr[6] = 11;
    
    build(1, 1, n);
    
    printf("区间[2,5]的和: %d\n", query(1, 1, n, 2, 5));  // 24
    
    update(1, 1, n, 3, 10);  // 把第3个元素改成10
    printf("修改后[2,5]的和: %d\n", query(1, 1, n, 2, 5));  // 29
}

模板：区间修改（懒标记Lazy Propagation） 

typedef struct {
    int sum;
    int lazy;  // 懒标记
} Node;

Node tree[4 * MAXN];

// 下传懒标记
void pushDown(int node, int l, int r) {
    if (tree[node].lazy != 0) {
        int mid = (l + r) / 2;
        int leftLen = mid - l + 1;
        int rightLen = r - mid;
        
        // 传给左儿子
        tree[node * 2].sum += tree[node].lazy * leftLen;
        tree[node * 2].lazy += tree[node].lazy;
        
        // 传给右儿子
        tree[node * 2 + 1].sum += tree[node].lazy * rightLen;
        tree[node * 2 + 1].lazy += tree[node].lazy;
        
        tree[node].lazy = 0;  // 清空当前标记
    }
}

// 区间修改：给[ql, qr]每个元素加上val
void rangeUpdate(int node, int l, int r, int ql, int qr, int val) {
    if (ql <= l && r <= qr) {
        tree[node].sum += val * (r - l + 1);
        tree[node].lazy += val;
        return;
    }
    pushDown(node, l, r);
    
    int mid = (l + r) / 2;
    if (ql <= mid) rangeUpdate(node * 2, l, mid, ql, qr, val);
    if (qr > mid) rangeUpdate(node * 2 + 1, mid + 1, r, ql, qr, val);
    
    tree[node].sum = tree[node * 2].sum + tree[node * 2 + 1].sum;
}

// 区间查询（带懒标记版本）
int rangeQuery(int node, int l, int r, int ql, int qr) {
    if (ql <= l && r <= qr) return tree[node].sum;
    if (r < ql || l > qr) return 0;
    
    pushDown(node, l, r);
    
    int mid = (l + r) / 2;
    int leftSum = rangeQuery(node * 2, l, mid, ql, qr);
    int rightSum = rangeQuery(node * 2 + 1, mid + 1, r, ql, qr);
    return leftSum + rightSum;
}
